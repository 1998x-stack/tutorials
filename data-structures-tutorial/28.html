综合项目：实现一个Mini Redis

理论说了很多，让我们来做点真正的东西。我们要用 Python 实现一个简化版的 Redis——我称之为 MiniRedis。

项目背景与目标

Redis 之所以被称为"数据结构服务器"，是因为它的五种核心数据类型对应了五种经典数据结构。我们的 MiniRedis 将实现其中三种最具代表性的：

  • 字符串（String） → 用 dict 实现基本 KV 存储
  • 有序集合（Sorted Set） → 用跳表（Skip List）实现 ZADD/ZRANGE
  • 过期机制（TTL） → 用最小堆（heap）管理超时

额外实现 AOF 持久化——在宕机后能从日志文件恢复数据。

跳表实现（ZSet 核心）

import random
import heapq
import time
import threading

# ========== 跳表 —— 支撑 ZSet 的有序结构 ==========
class SkipListNode:
    def __init__(self, score, member, level):
        self.score = score
        self.member = member
        self.forward = [None] * (level + 1)  # 多层指针

class SkipList:
    MAX_LEVEL = 16
    P = 0.5  # 层数增长概率

    def __init__(self):
        self.head = SkipListNode(-float('inf'), None, self.MAX_LEVEL)
        self.level = 0

    def random_level(self):
        lvl = 0
        while random.random() < self.P and lvl < self.MAX_LEVEL:
            lvl += 1
        return lvl

    def insert(self, score, member):
        """ZADD：插入或更新成员分数"""
        update = [None] * (self.MAX_LEVEL + 1)
        curr = self.head
        # 从顶层向下，找到每层的前驱
        for i in range(self.level, -1, -1):
            while curr.forward[i] and (
                curr.forward[i].score < score or
                (curr.forward[i].score == score and
                 curr.forward[i].member < member)
            ):
                curr = curr.forward[i]
            update[i] = curr
        curr = curr.forward[0]
        # 如果成员已存在，更新分数（简化：这里不处理删除原键）
        if curr and curr.member == member:
            curr.score = score
            return
        # 插入新节点
        lvl = self.random_level()
        if lvl > self.level:
            for i in range(self.level + 1, lvl + 1):
                update[i] = self.head
            self.level = lvl
        new_node = SkipListNode(score, member, lvl)
        for i in range(lvl + 1):
            new_node.forward[i] = update[i].forward[i]
            update[i].forward[i] = new_node

    def range_by_score(self, min_score, max_score):
        """ZRANGE BY SCORE：按分数范围获取成员"""
        result = []
        curr = self.head.forward[0]
        while curr and curr.score <= max_score:
            if curr.score >= min_score:
                result.append((curr.member, curr.score))
            curr = curr.forward[0]
        return result

    def range_by_rank(self, start, end):
        """ZRANGE：按排名范围获取成员（0-indexed）"""
        curr = self.head.forward[0]
        idx = 0
        result = []
        while curr and idx <= end:
            if idx >= start:
                result.append((curr.member, curr.score))
            idx += 1
            curr = curr.forward[0]
        return result

核心引擎 + TTL + AOF

class MiniRedis:
    def __init__(self, aof_path="miniredis.aof"):
        self.data = {}            # 哈希表 —— String 存储
        self.zsets = {}           # dict of SkipList —— 有序集合
        self.ttl_heap = []        # 最小堆 [(过期时间戳, key), ...]
        self.ttl_map = {}         # 记录每个 key 的 TTL 值，方便删除
        self.aof_path = aof_path
        self.lock = threading.Lock()
        self._load_aof()

    # ---------- String 操作 ----------
    def set(self, key, value):
        with self.lock:
            self.data[key] = str(value)
            self._append_aof(f"SET {key} {value}")

    def get(self, key):
        with self.lock:
            self._expire_keys()
            return self.data.get(key)

    def delete(self, key):
        with self.lock:
            self.data.pop(key, None)
            # 清理 TTL
            if key in self.ttl_map:
                # 标记删除：不用实际移除堆中元素，以 ttl_map 为准
                del self.ttl_map[key]
            self._append_aof(f"DEL {key}")

    # ---------- ZSet 操作 ----------
    def zadd(self, zset_name, score, member):
        with self.lock:
            if zset_name not in self.zsets:
                self.zsets[zset_name] = SkipList()
            self.zsets[zset_name].insert(float(score), member)
            self._append_aof(f"ZADD {zset_name} {score} {member}")

    def zrange(self, zset_name, start, end):
        with self.lock:
            if zset_name not in self.zsets:
                return []
            return self.zsets[zset_name].range_by_rank(start, end)

    def zrange_by_score(self, zset_name, min_s, max_s):
        with self.lock:
            if zset_name not in self.zsets:
                return []
            return self.zsets[zset_name].range_by_score(min_s, max_s)

    # ---------- TTL 机制（最小堆） ----------
    def expire(self, key, seconds):
        """设置 key 的过期时间（秒）"""
        with self.lock:
            expiry = time.time() + seconds
            heapq.heappush(self.ttl_heap, (expiry, key))
            self.ttl_map[key] = expiry

    def _expire_keys(self):
        """惰性删除：在每次操作前检查并删除过期 key"""
        now = time.time()
        while self.ttl_heap and self.ttl_heap[0][0] <= now:
            expiry, key = heapq.heappop(self.ttl_heap)
            # 检查是否还有效（可能被重复设置或删除）
            if self.ttl_map.get(key) == expiry:
                self.data.pop(key, None)
                del self.ttl_map[key]

    # ---------- AOF 持久化 ----------
    def _append_aof(self, command):
        """将写命令追加到 AOF 日志"""
        with open(self.aof_path, "a") as f:
            f.write(command + "\n")

    def _load_aof(self):
        """启动时重放 AOF 日志，恢复数据"""
        try:
            with open(self.aof_path, "r") as f:
                for line in f:
                    line = line.strip()
                    if not line:
                        continue
                    parts = line.split()
                    if parts[0] == "SET" and len(parts) >= 3:
                        self.data[parts[1]] = " ".join(parts[2:])
                    elif parts[0] == "DEL" and len(parts) == 2:
                        self.data.pop(parts[1], None)
                    elif parts[0] == "ZADD" and len(parts) == 4:
                        zset_name, score, member = parts[1], float(parts[2]), parts[3]
                        if zset_name not in self.zsets:
                            self.zsets[zset_name] = SkipList()
                        self.zsets[zset_name].insert(score, member)
        except FileNotFoundError:
            pass  # 首次启动，没有 AOF 文件

# ========== 使用示例 ==========
if __name__ == "__main__":
    r = MiniRedis()
    r.set("name", "小明")
    r.set("score", 98)
    print(r.get("name"))      # 输出：小明

    r.expire("temp_data", 2)
    time.sleep(3)
    print(r.get("temp_data")) # 输出：None（已过期）

    # 有序集合
    r.zadd("leaderboard", 85.5, "Alice")
    r.zadd("leaderboard", 92.0, "Bob")
    r.zadd("leaderboard", 78.0, "Charlie")
    print(r.zrange("leaderboard", 0, -1))
    # 输出：[('Charlie', 78.0), ('Alice', 85.5), ('Bob', 92.0)]

    # 按分数范围查询
    print(r.zrange_by_score("leaderboard", 80, 100))
    # 输出：[('Alice', 85.5), ('Bob', 92.0)]

项目扩展挑战

上面的实现约 150 行代码，已经是一个可工作的 MEMCACHED/REDIS 子集。如果你想继续深入，可以尝试以下扩展：

  • 网络协议：用 Python 的 socket 实现 RESP（Redis 序列化协议），让 MiniRedis 接受 TCP 连接。
  • RDB 快照：除了 AOF，实现定期全量快照（序列化整个 data dict 到文件）。
  • 并发控制：将 threading.Lock 替换为 multiprocessing 或 asyncio，提升并发性能。
  • 内存优化：为小字符串实现压缩编码（类似 Redis 的 embstr/int 编码优化）。
Mini Redis 项目用到的数据结构：哈希表（dict）、跳表（Skip List）、最小堆（heapq）、字符串——第 3、7、11、22 章的知识全部用上了。这就是数据结构的"实战"——不是孤立的考点，而是组合在一起的工程工具。你每学一个结构，都是在往你的"工具箱"里加一件趁手的工具。

这些变化不是"学会了 28 个知识点"，而是思维方式的重塑。数据结构不是一门"学完就忘"的课——它是一种伴随你整个编程生涯的底层思维。无论你以后做前端、后端、数据科学、AI 工程，数据结构都会像空气一样，无处不在却不可或缺。

最后，想分享一个我自己的故事。我学数据结构的启蒙老师在第一节课上说了一句话，我一直记到今天：

"数据结构很简单——就是两样东西：数组和指针。所有数据结构都是数组和指针的排列组合。"

这句话有些夸张，但不无道理。数组代表"连续"，指针代表"离散"。数组和指针组成了链表（连续 + 离散），链表延伸出了树（多个指针），树延伸出了图（允许成环）。哈希表是数组 + 链表，堆是特殊的数组，跳表是多层链表……所有结构都是基本元素的组合。

当你理解了这一点，你就不再是"学会"了数据结构——你内化了数据结构。

数据结构的学习没有终点——你永远会发现新的数据结构、新的应用场景、新的优化角度。这门课只是一个起点。现在，带着你学到的知识和思维方式，去写代码、去解决真实的问题、去创造属于你自己的数据结构吧。
