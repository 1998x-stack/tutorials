20.1 PyTorch：深度学习的事实标准

2026年，PyTorch已经是深度学习领域毫无争议的王者。它不是在"和TensorFlow竞争"——它已经赢了。学术界近90%的论文用PyTorch实现，工业界的采用率也超过了80%。这一章我们会从零开始，用PyTorch搭建一个完整的神经网络训练pipeline。

PyTorch的哲学：Define-by-Run——计算图在你每次前向传播时动态构建。这意味着你可以用Python原生的if/for/debug，不像TensorFlow 1.x那样需要先定义静态图再执行。这种灵活性对于研究和调试来说是无价的。

20.2 Tensor基础：PyTorch的"NumPy on GPU"

import torch
import numpy as np

# 从各种来源创建Tensor
a = torch.tensor([1, 2, 3])                 # 从list
b = torch.from_numpy(np.array([4, 5, 6]))   # 从numpy
c = torch.randn(3, 4)                        # 随机正态分布
d = torch.zeros(2, 5)                        # 全零

# GPU操作
if torch.cuda.is_available():
    a = a.to('cuda')          # 移到GPU
    a = a.to('cpu')           # 移回CPU
    device = torch.device('cuda')

# Tensor属性
print(f"形状: {c.shape}, 数据类型: {c.dtype}, 设备: {c.device}")

# 梯度追踪
x = torch.tensor([2.0], requires_grad=True)
y = x ** 2 + 3 * x + 1
y.backward()
print(f"dy/dx at x=2: {x.grad}")  # 2*2 + 3 = 7.0
💡 autograd的原理：当你设置requires_grad=True时，PyTorch会在每一步操作时记录计算图。.backward()调用时，从输出节点沿图反向遍历，用链式法则自动计算梯度。这是整个深度学习训练的基石，而你几乎不需要手动干预——这就是框架的力量。

20.3 DataLoader：高效的数据管道

from torch.utils.data import Dataset, DataLoader
from sklearn.datasets import make_classification

class TabularDataset(Dataset):
    def __init__(self, X, y=None):
        self.X = torch.FloatTensor(X)
        self.y = torch.FloatTensor(y) if y is not None else None

    def __len__(self):
        return len(self.X)

    def __getitem__(self, idx):
        if self.y is not None:
            return self.X[idx], self.y[idx]
        return self.X[idx]

# DataLoader自动做batch、shuffle、多进程加载
dataset = TabularDataset(X_train, y_train)
loader = DataLoader(dataset, batch_size=64, shuffle=True,
                    num_workers=2, pin_memory=True)

20.4 定义模型：nn.Module的三种写法

import torch.nn as nn

class MLP(nn.Module):
    def __init__(self, input_dim, hidden_dims, output_dim, dropout=0.3):
        super().__init__()
        layers = []
        prev_dim = input_dim
        for h_dim in hidden_dims:
            layers.extend([
                nn.Linear(prev_dim, h_dim),
                nn.BatchNorm1d(h_dim),
                nn.ReLU(),
                nn.Dropout(dropout)
            ])
            prev_dim = h_dim
        layers.append(nn.Linear(prev_dim, output_dim))
        self.network = nn.Sequential(*layers)

    def forward(self, x):
        return self.network(x)

model = MLP(input_dim=30, hidden_dims=[128, 64, 32], output_dim=1)
print(f"参数量: {sum(p.numel() for p in model.parameters()):,}")

20.5 训练循环：深度学习项目的"心脏"

def train_epoch(model, loader, optimizer, criterion, device):
    model.train()
    total_loss = 0
    for X, y in loader:
        X, y = X.to(device), y.to(device)
        optimizer.zero_grad()
        output = model(X).squeeze()
        loss = criterion(output, y)
        loss.backward()
        optimizer.step()
        total_loss += loss.item() * X.size(0)
    return total_loss / len(loader.dataset)

def evaluate(model, loader, criterion, device):
    model.eval()
    total_loss, correct = 0, 0
    with torch.no_grad():
        for X, y in loader:
            X, y = X.to(device), y.to(device)
            output = model(X).squeeze()
            loss = criterion(output, y)
            total_loss += loss.item() * X.size(0)
            pred = (torch.sigmoid(output) > 0.5).float()
            correct += (pred == y).sum().item()
    return total_loss / len(loader.dataset), correct / len(loader.dataset)

# 主训练循环
device = torch.device('cuda' if torch.cuda.is_available() else 'cpu')
model = MLP(30, [128, 64, 32], 1).to(device)
criterion = nn.BCEWithLogitsLoss()
optimizer = torch.optim.AdamW(model.parameters(), lr=1e-3, weight_decay=0.01)
scheduler = torch.optim.lr_scheduler.CosineAnnealingLR(optimizer, T_max=50)

best_val_loss = float('inf')
for epoch in range(50):
    train_loss = train_epoch(model, train_loader, optimizer, criterion, device)
    val_loss, val_acc = evaluate(model, val_loader, criterion, device)
    scheduler.step()

    if val_loss < best_val_loss:
        best_val_loss = val_loss
        torch.save(model.state_dict(), 'best_model.pt')

    print(f"Epoch {epoch+1:3d}: train_loss={train_loss:.4f}, "
          f"val_loss={val_loss:.4f}, val_acc={val_acc:.4f}")
⚠️ 训练循环中的常见bug：(1) 忘记optimizer.zero_grad()——梯度会累积而不是替换。(2) 验证时忘记model.eval()——Dropout和BN在训练/推理模式下行为不同，torch.no_grad()还不够。(3) BCEWithLogitsLoss自带Sigmoid，模型的输出不要手动加Sigmoid——否则会双重Sigmoid。(4) 在GPU上创建的target和model输出数据类型不匹配——这会导致静默的错误结果。

20.6 断点续训与混合精度

# 保存完整训练状态
checkpoint = {
    'epoch': epoch,
    'model_state_dict': model.state_dict(),
    'optimizer_state_dict': optimizer.state_dict(),
    'scheduler_state_dict': scheduler.state_dict(),
    'best_val_loss': best_val_loss,
}
torch.save(checkpoint, 'checkpoint.pt')

# 恢复训练
checkpoint = torch.load('checkpoint.pt')
model.load_state_dict(checkpoint['model_state_dict'])
optimizer.load_state_dict(checkpoint['optimizer_state_dict'])

# 混合精度训练（省显存+加速）
scaler = torch.amp.GradScaler('cuda')
with torch.amp.autocast('cuda'):
    output = model(X)
    loss = criterion(output, y)
scaler.scale(loss).backward()
scaler.step(optimizer)
scaler.update()
