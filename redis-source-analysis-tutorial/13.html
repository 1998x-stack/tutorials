第13章：事件循环（上）：文件事件

大家好，我是你们的讲师。前面 12 章我们一直在讨论 Redis 的数据结构和命令实现，但有一个核心问题始终没有回答：当数千个客户端同时连接 Redis 时，Redis 是怎么管理所有这些连接的？它是怎么知道某个客户端发来了新命令？又是怎样在单线程中同时处理网络 IO 和命令执行的？

答案就在 Redis 的事件循环中。事件循环（Event Loop）是 Redis 的"心脏"——它驱动着整个服务器的运转。从这一章开始，我们将用两章的篇幅深入 ae.c 和 ae_epoll.c，拆解这个精巧的 Reactor 模式实现。本章聚焦文件事件（网络 IO），下一章讨论时间事件（定时任务）。

Reactor 模式概述

Redis 采用的是经典的 Reactor 模式：单线程 + 非阻塞 IO + IO 多路复用。所谓 Reactor，就是一个"反应堆"——它不主动发起 IO，而是把文件描述符（FD）注册到内核，告诉内核"当这个 FD 可读/可写时通知我"，然后自己阻塞在 epoll_wait 上等待通知。

特性 Reactor（Redis） Proactor（如 Windows IOCP）
IO 操作谁执行 应用程序（用户态） 操作系统（内核态）
通知方式 "可以读了" "已经读好了"
编程模型 同步非阻塞 异步
系统调用 epoll_wait + read/write io_submit（AIO）
Redis 是否使用 是 否

Redis 选择 Reactor 的原因很直接：单线程模型下，如果用了 Proactor，异步 IO 的回调依然要在事件循环中执行，并没有本质的区别。而 Reactor 的代码更简单、更可控、更易于调试。

aeEventLoop 结构

所有事件管理的核心是 aeEventLoop 结构体，定义在 ae.h 中：

/* ae.h */
typedef struct aeEventLoop {
    int maxfd;                 /* 当前注册的最大 FD */
    int setsize;               /* 事件表大小（可监视的最大 FD 数） */
    long long timeEventNextId; /* 时间事件 ID 生成器 */
    time_t lastTime;           /* 上次处理时间事件的时间 */
    aeFileEvent *events;       /* 文件事件表，按 FD 索引 */
    aeFiredEvent *fired;       /* 就绪事件表，epoll_wait 返回后填充 */
    aeTimeEvent *timeEventHead;/* 时间事件链表头 */
    int stop;                  /* 停止标志 */
    void *apidata;             /* 后端多路复用特有数据（如 epoll 的 fd） */
    aeBeforeSleepProc *beforesleep; /* 每次等待前调用的钩子 */
    aeBeforeSleepProc *aftersleep;  /* 每次等待后调用的钩子 */
    int flags;                 /* 事件循环标志 */
} aeEventLoop;

核心字段的含义：

  • events —— 一个数组，大小为 setsize，每个 FD 对应一个 aeFileEvent，记录了该 FD 注册的回调函数和掩码。
  • fired —— 同样大小的数组，aeApiPoll（即 epoll_wait）返回后，把就绪的 FD 和事件掩码填入此数组。
  • apidata —— 多态后端的私有数据。以 epoll 为例，它包含一个 epollfd 和 struct epoll_event 数组。
  • beforesleep / aftersleep —— 每次事件循环迭代中，aeApiPoll 调用前/后执行的钩子函数。这是 Redis 实现"延迟处理"的关键机制。

文件事件注册与触发

aeCreateFileEvent / aeDeleteFileEvent

/* ae.c */
int aeCreateFileEvent(aeEventLoop *eventLoop, int fd, int mask,
        aeFileProc *proc, void *clientData) {
    if (fd >= eventLoop->setsize) return AE_ERR;

    aeFileEvent *fe = &eventLoop->events[fd];

    /* 调用后端 API 注册到 epoll/kqueue */
    if (aeApiAddEvent(eventLoop, fd, mask) == -1)
        return AE_ERR;

    fe->mask |= mask;
    if (mask & AE_READABLE) fe->rfileProc = proc;
    if (mask & AE_WRITABLE) fe->wfileProc = proc;
    fe->clientData = clientData;

    if (fd > eventLoop->maxfd) eventLoop->maxfd = fd;
    return AE_OK;
}

文件事件的掩码有三种：AE_READABLE（FD 可读时触发）、AE_WRITABLE（FD 可写时触发）、AE_BARRIER（写优先——即使可读事件也先触发写回调，用于确保响应在读取下一个请求前被写出）。

AE_BARRIER 的作用：Redis 在 postponeClientRead 机制中用到 AE_BARRIER。它的作用是：在每次事件循环迭代中，先处理之前积压的写事件（把响应发出去），再去读新的请求。这避免了"读完新请求→处理→产生新响应→下次迭代才能写出"的不必要延迟。

epoll 后端源码

Redis 的事件循环设计了多态后端：Linux 上优先使用 epoll，macOS/FreeBSD 上使用 kqueue，兜底使用 select。多态通过 aeApiCreate / aeApiPoll / aeApiAddEvent / aeApiDelEvent 四个函数实现。

ae_epoll.c 源码解析

/* ae_epoll.c */
typedef struct aeApiState {
    int epfd;                           /* epoll 实例的 FD */
    struct epoll_event *events;         /* epoll_wait 返回的事件数组 */
} aeApiState;

static int aeApiCreate(aeEventLoop *eventLoop) {
    aeApiState *state = zmalloc(sizeof(aeApiState));
    state->events = zmalloc(sizeof(struct epoll_event) * eventLoop->setsize);
    state->epfd = epoll_create(1024);   /* 创建 epoll 实例 */
    eventLoop->apidata = state;
    return 0;
}

static int aeApiAddEvent(aeEventLoop *eventLoop, int fd, int mask) {
    aeApiState *state = eventLoop->apidata;
    struct epoll_event ee = {0};

    /* 如果 fd 已经注册了某些事件，使用 EPOLL_CTL_MOD 修改 */
    int op = eventLoop->events[fd].mask == AE_NONE ?
             EPOLL_CTL_ADD : EPOLL_CTL_MOD;

    ee.events = 0;
    if (mask & AE_READABLE)  ee.events |= EPOLLIN;
    if (mask & AE_WRITABLE)  ee.events |= EPOLLOUT;
    ee.data.fd = fd;
    epoll_ctl(state->epfd, op, fd, &ee);
    return 0;
}

static int aeApiPoll(aeEventLoop *eventLoop, struct timeval *tvp) {
    aeApiState *state = eventLoop->apidata;
    int retval, j;

    /* 阻塞在 epoll_wait 上，等待事件发生 */
    retval = epoll_wait(state->epfd, state->events, eventLoop->setsize,
                        tvp ? (tvp->tv_sec * 1000 + tvp->tv_usec / 1000) : -1);

    /* 将就绪事件转换为 aeFiredEvent */
    for (j = 0; j < retval; j++) {
        struct epoll_event *e = &state->events[j];
        eventLoop->fired[j].fd = e->data.fd;
        eventLoop->fired[j].mask = 0;
        if (e->events & EPOLLIN)  eventLoop->fired[j].mask |= AE_READABLE;
        if (e->events & EPOLLOUT) eventLoop->fired[j].mask |= AE_WRITABLE;
        if (e->events & EPOLLERR) eventLoop->fired[j].mask |= AE_WRITABLE | AE_READABLE;
    }
    return retval;
}

epoll 相比 select 的两大优势：第一，不传整个 FD 集给内核，避免了 O(N) 的复制；第二，就绪事件是直接返回的，无需 O(N) 遍历查找。在数千个连接但只有少量活跃连接的场景下，epoll 的性能优势是决定性的。

io_uring 后端（Redis 8.4 新特性）

Redis 8.4 引入了对 io_uring 的支持，这是 Linux 5.1 引入的异步 IO 框架。与 epoll 不同，io_uring 通过 SQ（Submission Queue） 和 CQ（Completion Queue） 两个环形缓冲区实现内核与用户态的通信，减少了系统调用的次数。

/* ae_uring.c —— 简化示意 */
static int aeApiPoll(aeEventLoop *eventLoop, struct timeval *tvp) {
    struct io_uring_cqe *cqe;
    unsigned head;
    unsigned count = 0;

    /* 等待 CQ 中的完成事件 */
    io_uring_submit_and_wait(&state->ring, 1);

    /* 遍历 CQ 获取就绪的 IO */
    io_uring_for_each_cqe(&state->ring, head, cqe) {
        eventLoop->fired[count].fd = cqe->user_data;
        eventLoop->fired[count].mask = /* 转换 cqe->res 为 AE 掩码 */;
        count++;
    }
    io_uring_cq_advance(&state->ring, count);
    return count;
}
踩坑实录：epoll_wait 的超时参数是事件循环中一个容易被忽略的细节。超时设置太长，会导致时间事件（如清理过期键）无法及时执行；超时设置太短（比如 0），又会导致 CPU 空转——空转时每秒数十万次 epoll_wait 立即返回。Redis 的 aeProcessEvents 会根据最近一个时间事件的触发时间来计算 epoll_wait 的超时：如果有即将到期的时间事件，超时设为到那个时间点的间隔；如果没有，使用默认的常超时。但如果时间事件处理本身有延迟，这个计算就变得关键了。曾经有团队把 hz 调到 1000（每秒 1000 次 serverCron），结果 epoll_wait 每次只等 1ms，CPU 使用率从 10% 飙到了 60%。

事件处理主循环

所有事件处理逻辑的顶层入口是 aeMain：

/* ae.c */
void aeMain(aeEventLoop *eventLoop) {
    eventLoop->stop = 0;
    while (!eventLoop->stop) {
        /* 每次迭代前调用钩子 */
        if (eventLoop->beforesleep != NULL)
            eventLoop->beforesleep(eventLoop);

        /* 处理所有待办事件 */
        aeProcessEvents(eventLoop, AE_ALL_EVENTS | AE_CALL_BEFORE_SLEEP | AE_CALL_AFTER_SLEEP);

        /* 每次迭代后调用钩子 */
        if (eventLoop->aftersleep != NULL)
            eventLoop->aftersleep(eventLoop);
    }
}

int aeProcessEvents(aeEventLoop *eventLoop, int flags) {
    int processed = 0;

    /* 1. 计算 epoll_wait 的超时时间 */
    if (eventLoop->maxfd != -1 ||
        ((flags & AE_TIME_EVENTS) && !(flags & AE_DONT_WAIT))) {
        struct timeval tv, *tvp;
        int64_t usUntilTimer = -1;

        /* 获取最近一个时间事件的到期时间 */
        if (flags & AE_TIME_EVENTS && !(flags & AE_DONT_WAIT))
            usUntilTimer = usUntilEarliestTimer(eventLoop);

        if (usUntilTimer >= 0) {
            tv.tv_sec = usUntilTimer / 1000000;
            tv.tv_usec = usUntilTimer % 1000000;
            tvp = &tv;
        } else {
            if (flags & AE_DONT_WAIT) { tvp = NULL; } /* 不等待 */
            else { tvp = NULL; } /* 无限等待 */
        }

        /* 2. 调用 epoll_wait 等待文件事件 */
        numevents = aeApiPoll(eventLoop, tvp);
    }

    /* 3. 处理所有就绪的文件事件 */
    for (j = 0; j < numevents; j++) {
        aeFileEvent *fe = &eventLoop->events[fired->fd];

        /* 写优先：如果有 AE_BARRIER，先处理写事件 */
        if (fired->mask & AE_WRITABLE && (!(fired->mask & AE_READABLE) ||
            (fe->mask & AE_BARRIER))) {
            fe->wfileProc(eventLoop, fired->fd, fe->clientData, fired->mask);
            processed++;
        }

        /* 然后处理读事件 */
        if (fired->mask & AE_READABLE) {
            fe->rfileProc(eventLoop, fired->fd, fe->clientData, fired->mask);
            processed++;
        }

        /* 如果没设 AE_BARRIER，再写 */
        if (fired->mask & AE_WRITABLE && !(fe->mask & AE_BARRIER)) {
            fe->wfileProc(eventLoop, fired->fd, fe->clientData, fired->mask);
            processed++;
        }
    }

    /* 4. 处理时间事件 */
    if (flags & AE_TIME_EVENTS)
        processed += processTimeEvents(eventLoop);

    return processed;
}
单线程模型的真实含义：Redis 的"单线程"指的不是整个进程只有一个线程，而是"事件循环在主线程中串行处理所有请求"。从 Redis 6.0 开始，网络 IO 的读写操作可以委托给 IO 线程池并行执行，但命令执行（即 rfileProc 中调用的 processCommand 到具体命令函数）始终在主线程中串行执行。所以 SET/GET 命令的执行仍然是线程安全的，无需加锁。

文件事件驱动着 Redis 的网络 IO——每次有客户端发送命令，epoll 都会通知 Redis，然后 Redis 读取数据、解析命令、执行、返回结果。但除了网络 IO，Redis 还需要做定期任务：清理过期键、检查 AOF 重写、发送复制心跳……这些任务由时间事件驱动。下一章我们继续分析时间事件的实现和 serverCron 的细节。
