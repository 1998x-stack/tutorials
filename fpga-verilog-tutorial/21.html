第21章 RISC-V处理器FPGA实现

2010 年，UC Berkeley 的 David Patterson 团队发布了一套全新的指令集架构——RISC-V。与 ARM 和 x86 不同，RISC-V 是开放、免费的，其设计哲学是"简洁、模块化、可扩展"。你不需要支付任何授权费就能在自己的 FPGA 上跑一个 RISC-V 处理器。2017 年我教计算机体系结构课程时，学生们用手头的 Basys3 开发板在实验室里实现了自己写的 RISC-V 核并成功跑通了 C 程序——这个"从零到 CPU"的成就感，是单纯理论讲解无法替代的。

本章我们将实现一个 RV32I 的五级流水线 RISC-V 核，并在真实的 FPGA 上运行 C 程序。

21.1 RV32I 指令集简介

RV32I 是 RISC-V 的基础整数指令集，包含约 40 条指令。指令格式被精简为六种统一的编码类型：

指令格式opcodefunct3funct7描述
ADD rd, rs1, rs2R01100110000000000rs1 + rs2 → rd
SUB rd, rs1, rs2R01100110000100000rs1 - rs2 → rd
AND rd, rs1, rs2R01100111110000000rs1 & rs2 → rd
OR rd, rs1, rs2R01100111100000000rs1 | rs2 → rd
XOR rd, rs1, rs2R01100111000000000rs1 ^ rs2 → rd
ADDI rd, rs1, immI0010011000—rs1 + imm → rd
LW rd, offset(rs1)I0000011010—Load word from memory
SW rs2, offset(rs1)S0100011010—Store word to memory
BEQ rs1, rs2, offsetB1100011000—Branch if equal
JAL rd, offsetJ1101111——Jump and link (for function call)
LUI rd, immU0110111——Load upper immediate

RV32I 总共只有 六个格式（R/I/S/B/U/J），解码逻辑非常规整。所有指令都是 32 位定长编码。寄存器堆有 32 个通用寄存器（x0~x31），其中 x0 硬连到 0。这种简洁性让 RISC-V 特别适合教学和FPGA 实现——你不需要解码复杂的变长指令或处理大量寻址模式。

21.2 五级流水线 RISC-V 核设计

我们的处理器采用经典的五级流水线结构：IF（取指）→ ID（译码与寄存器读取）→ EX（执行）→ MEM（访存）→ WB（写回）。下面是 ALU 模块的实现——它在 EX 阶段完成所有整数运算：

// RV32I ALU Module — 完整译码与执行单元
module rv32i_alu (
    input  wire [31:0] src_a,        // rs1 value
    input  wire [31:0] src_b,        // rs2 value or immediate
    input  wire [3:0]  alu_ctrl,     // ALU operation select
    output reg  [31:0] result,
    output reg         zero          // result == 0 flag (for BEQ/BNE)
);
    always @(*) begin
        case (alu_ctrl)
            4'b0000: result = src_a + src_b;                    // ADD / ADDI
            4'b0001: result = src_a - src_b;                    // SUB
            4'b0010: result = src_a & src_b;                    // AND / ANDI
            4'b0011: result = src_a | src_b;                    // OR  / ORI
            4'b0100: result = src_a ^ src_b;                    // XOR / XORI
            4'b0101: result = src_a << src_b[4:0];              // SLL / SLLI
            4'b0110: result = src_a >> src_b[4:0];              // SRL / SRLI
            4'b0111: result = $signed(src_a) >>> src_b[4:0];   // SRA / SRAI
            4'b1000: result = ($signed(src_a) < $signed(src_b))
                              ? 32'd1 : 32'd0;                  // SLT / SLTI
            4'b1001: result = (src_a < src_b) ? 32'd1 : 32'd0; // SLTU / SLTUI
            default:  result = 32'd0;
        endcase
        zero = (result == 32'd0);
    end
endmodule

下面的简化五级流水线顶层模块展示了数据通路的骨架。为了节省篇幅，省略了完整的前推多路选择器和冒险检测逻辑，但保留了最核心的流水线寄存器链：

// Simplified 5-Stage RV32I Pipeline Top
module rv32i_pipeline (
    input  wire        clk,
    input  wire        rst_n,
    // Instruction memory interface
    output reg  [31:0] pc,
    input  wire [31:0] instr,
    // Data memory interface
    output wire [31:0] dmem_addr,
    input  wire [31:0] dmem_rdata,
    output wire [31:0] dmem_wdata,
    output wire        dmem_we
);
    // Register File
    reg [31:0] rf [31:0];
    wire [4:0] rs1_addr = instr[19:15];
    wire [4:0] rs2_addr = instr[24:20];
    wire [4:0] rd_addr;

    // ======== IF/ID Pipeline Register ========
    reg [31:0] if_id_pc, if_id_instr;
    always @(posedge clk) begin
        if (!rst_n) begin
            if_id_pc    <= 32'h0;
            if_id_instr <= 32'h0;
            pc          <= 32'h0;
        end else begin
            pc          <= pc + 32'd4;
            if_id_pc    <= pc;
            if_id_instr <= instr;
        end
    end

    // ======== ID Stage ========
    wire [6:0]  opcode    = if_id_instr[6:0];
    wire [2:0]  funct3    = if_id_instr[14:12];
    wire [6:0]  funct7    = if_id_instr[31:25];
    reg  [3:0]  alu_ctrl;
    reg         reg_wr_en;
    reg         mem_we;
    reg         alu_src_b;  // 0=rs2, 1=imm
    // ... immediate extension, control decode ...

    // ======== ID/EX Pipeline Register ========
    reg [31:0] id_ex_pc;
    reg [31:0] id_ex_rs1, id_ex_rs2;
    reg [4:0]  id_ex_rd;
    reg [3:0]  id_ex_alu_ctrl;
    reg        id_ex_reg_wr, id_ex_mem_we, id_ex_alu_src_b;

    // ======== EX Stage ========
    wire [31:0] alu_src_b_mux = id_ex_alu_src_b ? imm_ext : id_ex_rs2;
    wire [31:0] alu_result;
    wire        alu_zero;
    rv32i_alu alu_inst (
        .src_a(id_ex_rs1),
        .src_b(alu_src_b_mux),
        .alu_ctrl(id_ex_alu_ctrl),
        .result(alu_result),
        .zero(alu_zero)
    );

    // ======== EX/MEM Pipeline Register ========
    reg [31:0] ex_mem_alu_out;
    reg [31:0] ex_mem_rs2;
    reg [4:0]  ex_mem_rd;
    reg        ex_mem_reg_wr, ex_mem_mem_we;

    // ======== MEM Stage ========
    assign dmem_addr  = ex_mem_alu_out;
    assign dmem_wdata = ex_mem_rs2;
    assign dmem_we    = ex_mem_mem_we;

    // ======== MEM/WB Pipeline Register ========
    reg [31:0] mem_wb_result;
    reg [31:0] mem_wb_mem_data;
    reg [4:0]  mem_wb_rd;
    reg        mem_wb_reg_wr;

    // ======== WB Stage ========
    always @(posedge clk) begin
        if (mem_wb_reg_wr && (mem_wb_rd != 0))
            rf[mem_wb_rd] <= mem_wb_result;
    end

endmodule
动手建议：上面的代码是一个骨架——缺少了控制单元译码、立即数扩展、前推（Forwarding）和冒险检测（Hazard Detection）。强烈建议在理解骨架的基础上，自己补全缺失的译码逻辑和前推多路选择器。这是计算机体系结构课程中最有价值的练习之一。完整实现大约需要 300~400 行 Verilog。

21.3 流水线冒险处理

流水线处理器有三个经典的冒险问题需要解决：

  • 数据冒险（Data Hazard）：后面的指令需要前面指令的运算结果。解决方法是数据前推（Forwarding / Bypassing）——把 EX、MEM、WB 阶段的结果直接转发到 EX 阶段的 ALU 输入端，而不是等写回寄存器堆再读。前推逻辑的关键是三组 Forwarding MUX，分别从 EX/MEM 和 MEM/WB 流水线寄存器取结果。
  • Load-Use 冒险：加载指令（LW）之后紧跟着使用加载结果的指令。即使有前推，LW 的结果在 EX 阶段结束前不可用，所以必须插入一个气泡（Stall / Bubble），把流水线冻结一拍。
  • 控制冒险（Control Hazard）：分支指令（BEQ/BNE）需要知道跳转是否发生。最简单的策略是静态预测不跳转——取指阶段先取下一条地址，如果分支实际发生，冲刷（Flush）已进入流水线的两条错误指令。

21.4 在 FPGA 上运行 RISC-V 程序

写完 RTL，如何验证它在 FPGA 上真的能运行 C 程序？典型流程如下：

  1. 用 RISC-V GCC 交叉编译工具链编译 C 程序：riscv32-unknown-elf-gcc -march=rv32i -O2 -o hello.elf hello.c。
  2. 用 riscv32-unknown-elf-objcopy -O verilog hello.elf hello.hex 将 ELF 文件转换为 Verilog 可加载的 hex 格式。
  3. 将 hello.hex 加载到 RTL 仿真模型的指令存储器中，运行仿真验证行为正确。
  4. 综合实现后在 FPGA 上运行，通过 UART 输出 "Hello, RISC-V!" 字符串。
踩坑——遗忘的数据前推：在我的第一个 RISC-V 实现中，我天真地认为流水线寄存器会"顺其自然地"把数据传下去。结果在仿真一个简单的冒泡排序程序时，排序结果完全是错的。调试了一整天，发现根因是连续两条 R-type 指令：ADD x1, x2, x3 之后紧接 SUB x4, x1, x5——EX 阶段读取的 x1 还是旧值，没有等到前一条指令的 ALU 结果写回。仿真通过了单条指令的测试，但因为特定指令对的出现才暴露了问题。修复方案就是在前推 MUX 中判断：如果 EX 阶段的 rs1/rs2 地址等于前一条（EX/MEM）或前前条（MEM/WB）的目标寄存器地址，就截取转发结果而不是寄存器堆输出。
