Congratulations——你已经遍历了信息论的核心版图。但这门学科最深的洞见往往来自于"动手做"。本章将带你完成三个层次递进的实战项目：从验证基础概念，到实现经典算法，再到构建端到端系统。

在开始之前，说一个我自己刚学信息论时踩过的坑。当时我自信满满地实现了Huffman编码，对一段英文文本做了压缩，一看结果——压缩比只有1.1:1，几乎没压缩。检查了半天发现：我忘了在编码前先统计输入的概率分布，而是假设每个字符等概率出现。这相当于你试图压缩一段"全零"的数据，却事先认为它可能是任何东西——那当然压缩不了。这件事教会我：信息论的起点永远是"了解你的数据"。不知道数据的分布，任何编码方案都是空中楼阁。

项目一：熵与互信息的可视化计算器

第一个项目从基础工具开始——构建一个可以交互式计算信息论核心量的程序。虽然我们已经在前面的章节中用Python片段做过类似的事情，但现在是时候把它做成一个完整、可复用的工具了。

这个工具需要支持：

  • 从离散概率分布计算：熵 H(X)、联合熵 H(X,Y)、条件熵 H(X|Y)、互信息 I(X;Y)、KL散度 DKL(P||Q)
  • 从样本数据（连续域）自动估计概率密度并计算上述量（使用核密度估计或k近邻方法）
  • 以信息Venn图的形式可视化熵、条件熵和互信息之间的关系

核心实现思路如下。对于离散分布，直接使用定义式计算即可。对于连续采样数据，我们使用Kraskov-Stoegbauer-Grassberger（KSG）估计器——这是一个基于k近邻距离的非参数互信息估计方法，不需要显式估计概率密度，在中等维度（d ≤ 10）下性能优良。

import numpy as np
from collections import Counter

def entropy(probs):
    """计算离散熵 H(X) = -Σ p(x) log₂ p(x)"""
    return -np.sum(probs * np.log2(probs + 1e-15))

def joint_entropy(joint_probs):
    """计算联合熵 H(X,Y)"""
    return -np.sum(joint_probs * np.log2(joint_probs + 1e-15))

def mutual_information(joint_probs, margin_x, margin_y):
    """计算互信息 I(X;Y)"""
    mi = 0.0
    for i in range(len(margin_x)):
        for j in range(len(margin_y)):
            if joint_probs[i,j] > 0:
                mi += joint_probs[i,j] * np.log2(
                    joint_probs[i,j] / (margin_x[i] * margin_y[j])
                )
    return mi

def kl_divergence(P, Q):
    """计算KL散度 D_KL(P||Q)"""
    return np.sum(P * np.log2((P + 1e-15) / (Q + 1e-15)))

# 示例：掷骰子游戏
p_x = np.array([1/6] * 6)  # 公平骰子
p_y = np.array([1/6] * 6)  # 另一个公平骰子

# 联合分布：独立
joint_indep = np.outer(p_x, p_y)
h_xy_joint = joint_entropy(joint_indep)
print(f"独立时 H(X) = {entropy(p_x):.3f} bits")
print(f"独立时 H(X,Y) = {h_xy_joint:.3f} bits")
print(f"独立时 I(X;Y) = {mutual_information(joint_indep, p_x, p_y):.3f} bits")

# 联合分布：完全相关（Y = X）
joint_corr = np.diag(p_x)  # 只有对角线非零
h_xy_corr = joint_entropy(joint_corr)
i_xy_corr = mutual_information(joint_corr, p_x, p_y)
print(f"完全相关时 H(X,Y) = {h_xy_corr:.3f} bits")
print(f"完全相关时 I(X;Y) = {i_xy_corr:.3f} bits")
# 期望输出：完全相关时互信息 = log₂(6) ≈ 2.585 bits

# 信息Venn图示意
print("\n信息Venn图 (完全相关):")
print(f"  ┌─────────────────────┐")
print(f"  │  H(X) = {entropy(p_x):.3f}        │")
print(f"  │    ┌───────────┐    │")
print(f"  │    │ I = {i_xy_corr:.3f}  │    │")
print(f"  │    └───────────┘    │")

第二个项目将我们学过的三种无损压缩算法——Huffman编码、算术编码和LZW字典编码——从零实现，并在不同数据上系统对比。

准备三个不同的测试数据集：

  • 英文文本：一篇典型的英文文章（如Shannon1948年的论文原文），约50KB
  • DNA序列：只含{A, C, G, T}四个字符的碱基序列，约100KB
  • 随机噪声：均匀随机字节，约10KB（作为baseline——理论上，随机数据不可压缩）
import heapq
from collections import Counter

class HuffmanCoding:
    """Huffman编码：最优前缀码"""

    class Node:
        def __init__(self, char, freq):
            self.char = char
            self.freq = freq
            self.left = None
            self.right = None
        def __lt__(self, other):
            return self.freq < other.freq

    def build_tree(self, text):
        """根据输入文本构建Huffman树"""
        freq = Counter(text)
        heap = [self.Node(char, f) for char, f in freq.items()]
        heapq.heapify(heap)

        while len(heap) > 1:
            left = heapq.heappop(heap)
            right = heapq.heappop(heap)
            merged = self.Node(None, left.freq + right.freq)
            merged.left = left
            merged.right = right
            heapq.heappush(heap, merged)

        return heap[0] if heap else None

    def build_codes(self, node, prefix="", code_map=None):
        """遍历Huffman树生成编码表"""
        if code_map is None:
            code_map = {}
        if node.char is not None:
            code_map[node.char] = prefix or "0"
        else:
            self.build_codes(node.left, prefix + "0", code_map)
            self.build_codes(node.right, prefix + "1", code_map)
        return code_map

项目三：端到端通信系统

第三个项目是最终的大项目——构建一个完整的端到端通信系统，将我们学过的所有知识串起来。系统架构如下：

                  ┌──────────┐    ┌──────────┐    ┌──────┐    ┌──────────┐    ┌──────────┐
原始数据 → → → → →│ 信源编码  │ → │ 信道编码  │ → │ 信道  │ → │ 信道译码  │ → │ 信源译码  │ → → → → → 恢复数据
                  │ Huffman  │    │ Hamming  │    │ BSC  │    │ 纠错译码  │    │ Huffman  │
                  │ 或算术编码 │    │ 或卷积码  │    │或AWGN│    │ Viterbi   │    │ 或算术译码 │
                  └──────────┘    └──────────┘    └──────┘    └──────────┘    └──────────┘
                                                     ↑
                                                 噪声源
                                                 (可控SNR)

这个系统最棒的地方在于：每一级都是一个你在之前章节中已经学过概念的独立模块。现在你要把它们组合在一起，观察整体性能。

from scipy import stats

class CommunicationSystem:
    """
    端到端通信系统骨架

    TODO 2: 实现信道编码器（Hamming或卷积码）

    def channel_decode(self, received_bits):

        return received_bits  # placeholder

    def transmit(self, bits):
        """通过信道传输：添加噪声"""
        if self.channel_type == "bsc":
            p_error = self._snr_to_bep(self.snr_db)
            noise = np.random.random(len(bits)) < p_error
            return bits ^ noise  # 翻转被噪声影响的比特
        elif self.channel_type == "awgn":
            snr_linear = 10 ** (self.snr_db / 10)
            noise_std = 1 / np.sqrt(2 * snr_linear)
            # 将bits映射为BPSK信号：0 → -1, 1 → +1
            signal = 2 * bits - 1
            received = signal + noise_std * np.random.randn(len(bits))
            return (received > 0).astype(int)
        return bits

    def _snr_to_bep(self, snr_db):
        """将SNR(dB)转换为BSC的比特错误概率"""
        # 对于AWGN下的BPSK: BEP = 0.5 * erfc(sqrt(SNR_linear))
        snr_linear = 10 ** (snr_db / 10)
        return 0.5 * stats.erfc(np.sqrt(snr_linear))

    def run_simulation(self, data):
        """运行一次完整的端到端仿真"""
        print(f"原始数据大小: {len(data)} bytes = {len(data)*8} bits")

        # 信源编码
        compressed = self.source_encode(data)
        print(f"压缩后大小: {len(compressed)} bits")
        print(f"压缩比: {len(data)*8 / len(compressed):.2f}x")

        # 信道编码
        coded = self.channel_encode(compressed)
        code_rate = len(compressed) / len(coded)
        print(f"信道编码率: {code_rate:.2f} (添加冗余: {len(coded)} bits)")

        # 传输（加噪声）
        received = self.transmit(coded)
        raw_ber = np.mean(coded != received)
        print(f"信道原始误码率: {raw_ber:.4f}")

        # 信道译码
        decoded_bits = self.channel_decode(received)

        # 信源译码
        reconstructed = self.source_decode(decoded_bits)
        final_ber = self._compute_ber(data, reconstructed)
        print(f"最终误码率: {final_ber:.6f}")

        return final_ber

    def _compute_ber(self, original, reconstructed):
        """计算重建数据的误码率"""
        if isinstance(original, str):
            original = original.encode()
        o_bits = np.unpackbits(np.frombuffer(original, dtype=np.uint8))
        r_bits = np.unpackbits(np.frombuffer(reconstructed, dtype=np.uint8))
        min_len = min(len(o_bits), len(r_bits))
        return np.mean(o_bits[:min_len] != r_bits[:min_len])

    def sweep_snr(self, data, snr_range_db):
        """扫掠SNR，绘制BER曲线"""
        ber_curve = []
        for snr in snr_range_db:
            self.snr_db = snr
            ber = self.run_simulation(data)
            ber_curve.append(ber)
        return ber_curve
