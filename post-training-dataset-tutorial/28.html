第28章 综合实战：从零构建后训练数据流水线

大家好，我是你们的讲师。欢迎来到本课程的最后一章。在过去的27章中，我们一起走过了一条完整的知识路径：从后训练数据的核心概念（第1-6章），到主流数据集的深入剖析（第7-14章），再到偏好对齐和推理增强数据（第15-22章），接着是工程体系和合规（第23-25章），最后是前沿趋势和策略设计（第26-27章）。

现在是时候把所有这些知识串起来了。这一章，我们将完成一个端到端的后训练数据流水线项目——从一个空白目录开始，经过数据集选择、下载、清洗、配比、偏好数据构造、格式转换，到最终产出可以直接用于训练的 JSON 文件。整个过程用一份完整的 Python 脚本实现。

这个项目的目标是为一个通用中文对话模型构建完整的 SFT + DPO 数据流水线。我们选择了四个经过验证的开源数据集作为输入。为什么选这四个？它们分别覆盖了通用能力、中文对话、中文指令和安全对齐四个维度——正如我们在第27章"中文模型"场景中推荐的方案。

Step 1: 数据集选择与下载

首先，我们需要一个统一的下载脚本。我们使用 HuggingFace datasets 库来加载所有数据集：

# download_datasets.py
from datasets import load_dataset
import json, os

os.makedirs("data/raw", exist_ok=True)

datasets_config = {
    "openhermes": {
        "path": "teknium/OpenHermes-2.5",
        "split": "train",
        "size": 100000
    },
    "ultrachat_zh": {
        "path": "stingning/ultrachat",
        "split": "train",
        "size": 50000
    },
    "firefly": {
        "path": "YeungNLP/firefly-train-1.1M",
        "split": "train",
        "size": 80000
    },
    "coig": {
        "path": "BAAI/COIG-PE",
        "split": "train",
        "size": 20000
    },
    "ultrafeedback": {
        "path": "openbmb/UltraFeedback",
        "split": "train",
        "size": 30000
    },
    "beavertails": {
        "path": "PKU-Alignment/BeaverTails",
        "split": "train",
        "size": 20000
    }
}

for name, cfg in datasets_config.items():
    print(f"Downloading {name}...")
    ds = load_dataset(
        cfg["path"],
        split=cfg["split"],
        streaming=True
    )
    samples = []
    for i, example in enumerate(ds):
        if i >= cfg["size"]:
            break
        samples.append(example)
    with open(f"data/raw/{name}.json", "w") as f:
        json.dump(samples, f, ensure_ascii=False)
    print(f"  → Saved {len(samples)} samples to data/raw/{name}.json")
下载建议：上面使用了 streaming=True，这在大数据集上可以避免内存溢出。如果你的网络条件允许，也可以设置 streaming=False 配合 cache_dir 参数实现本地缓存。第一次下载会比较慢，之后 HuggingFace 会自动缓存，第二次会快很多。

Step 2: 数据清洗与去重

不同数据集有不同的字段名和格式，我们需要统一处理。同时，我们要做两轮去重——数据集内部去重和跨数据集去重：

# clean_and_dedup.py
import json, re, hashlib
from collections import defaultdict

# MinHash 去重简化实现
def minhash_signature(text, num_hashes=128):
    """对文本计算 MinHash 签名"""
    shingles = set()
    for i in range(len(text) - 4):
        shingle = text[i:i+5]
        shingles.add(hash(shingle))
    # 取最小的 num_hashes 个 hash 值作为签名
    return sorted(list(shingles))[:num_hashes]

def jaccard_similarity(sig1, sig2):
    """估计两个 MinHash 签名的 Jaccard 相似度"""
    if not sig1 or not sig2:
        return 0.0
    intersection = len(set(sig1) & set(sig2))
    union = len(set(sig1) | set(sig2))
    return intersection / union if union > 0 else 0.0

def normalize_field(sample, field_mapping):
    """统一字段名"""
    normalized = {}
    for target_key, source_keys in field_mapping.items():
        for key in source_keys:
            if key in sample and sample[key]:
                normalized[target_key] = sample[key]
                break
        if target_key not in normalized:
            normalized[target_key] = ""
    return normalized

def quality_filter(sample, min_length=10, max_length=4096):
    """质量过滤：去除过短、过长、含明显噪声的样本"""
    text = sample.get("instruction", "") + " " + sample.get("output", "")
    if len(text) < min_length or len(text) > max_length:
        return False
    # 过滤重复字符比例过高的样本
    if max(re.findall(r'(.)\1{10,}', text)):
        return False
    # 过滤乱码
    garbage_patterns = [r'http[s]?://\S+[^\w\s]{5,}', r'[^\w\s]{20,}']
    for pat in garbage_patterns:
        if re.search(pat, text):
            return False
    return True

def deduplicate(datasets, threshold=0.7):
    """跨数据集去重"""
    all_signatures = {}
    cleaned = {}
    for name, samples in datasets.items():
        cleaned[name] = []
        for sample in samples:
            prompt = sample.get("instruction", "") or sample.get("prompt", "")
            sig = minhash_signature(prompt)
            is_dup = False
            for existing_sig in all_signatures.values():
                if jaccard_similarity(sig, existing_sig) > threshold:
                    is_dup = True
                    break
            if not is_dup:
                all_signatures[f"{name}_{len(cleaned[name])}"] = sig
                cleaned[name].append(sample)
    return cleaned

Step 3: 配比混合

根据第27章的中文模型配比方案，我们按 60/25/10/5 的比例混合四类数据：

# mix_and_sample.py
import json, random

def mix_datasets(cleaned_data, target_size=80000, seed=42):
    """
    按配比混合数据集
    配比: 通用 60% / 中文对话 25% / 推理 10% / 安全 5%
    """
    random.seed(seed)
    ratios = {
        "openhermes": 0.60,    # 通用能力
        "ultrachat_zh": 0.25,  # 中文对话
        "firefly": 0.10,       # 推理指令
        "coig": 0.05           # 安全对齐
    }

    mixed = []
    for name, ratio in ratios.items():
        samples = cleaned_data[name]
        n = int(target_size * ratio)
        # 如果该数据源样本不足，全量使用
        sampled = random.sample(samples, min(n, len(samples)))
        for s in sampled:
            s["source"] = name
        mixed.extend(sampled)
        print(f"  {name}: 需求{n}条，实际采样{len(sampled)}条")

    random.shuffle(mixed)
    print(f"  混合完成，共{len(mixed)}条样本")
    return mixed

Step 4: 偏好数据构造

DPO 数据需要 chosen 和 rejected 成对出现。我们从 UltraFeedback 和 BeaverTails 中构造出标准格式：

# build_preference.py
def build_dpo_pairs_from_ultrafeedback(samples):
    """从 UltraFeedback 构建 DPO 训练对"""
    dpo_pairs = []
    for sample in samples:
        prompt = sample.get("instruction", "")
        responses = sample.get("responses", [])
        ratings = sample.get("ratings", [])
        if len(responses) < 2 or len(ratings) < 2:
            continue
        # 选择评分最高的作为 chosen，最低的作为 rejected
        paired = list(zip(responses, ratings))
        paired.sort(key=lambda x: x[1], reverse=True)
        chosen = paired[0][0]
        rejected = paired[-1][0]
        dpo_pairs.append({
            "prompt": prompt,
            "chosen": chosen,
            "rejected": rejected,
            "source": "ultrafeedback"
        })
    return dpo_pairs

def build_dpo_pairs_from_beavertails(samples):
    """从 BeaverTails 构建安全偏好 DPO 对"""
    dpo_pairs = []
    for sample in samples:
        prompt = sample.get("prompt", "")
        safe_response = sample.get("safe_response", "")
        unsafe_response = sample.get("unsafe_response", "")
        if not safe_response or not unsafe_response:
            continue
        dpo_pairs.append({
            "prompt": prompt,
            "chosen": safe_response,
            "rejected": unsafe_response,
            "source": "beavertails"
        })
    return dpo_pairs
偏好数据的质量要点：构造 DPO 数据时，选择 chosen 和 rejected 的质量差距要足够明显。如果两条回答质量相近，DPO 的训练信号会很弱。在 UltraFeedback 中，建议只保留评分差距 ≥ 2 的样本（假设评分范围是1-5）。对于 BeaverTails，安全回答和不安全回答天然就有清晰的边界，很适合做 DPO 训练。

Step 5: 格式转换与导出

最后将 SFT 数据和 DPO 数据分别转换为训练框架兼容的格式：

# convert_format.py
import json

def to_sharegpt_format(sft_samples):
    """转换为 ShareGPT 格式（兼容 TRL, Axolotl）"""
    converted = []
    for s in sft_samples:
        conv = {
            "conversations": [
                {"from": "human", "value": s["instruction"]},
                {"from": "gpt", "value": s["output"]}
            ],
            "source": s.get("source", "unknown")
        }
        converted.append(conv)
    return converted

def to_hf_dpo_format(dpo_pairs):
    """转换为 HuggingFace DPO 格式"""
    converted = []
    for pair in dpo_pairs:
        converted.append({
            "prompt": pair["prompt"],
            "chosen": [{"role": "user", "content": pair["prompt"]},
                       {"role": "assistant", "content": pair["chosen"]}],
            "rejected": [{"role": "user", "content": pair["prompt"]},
                         {"role": "assistant", "content": pair["rejected"]}]
        })
    return converted

def save_datasets(sft_data, dpo_data, output_dir="data/final"):
    """保存最终数据集"""
    import os
    os.makedirs(output_dir, exist_ok=True)
    with open(f"{output_dir}/sft_data_sharegpt.json", "w") as f:
        json.dump(to_sharegpt_format(sft_data), f, ensure_ascii=False, indent=2)
    with open(f"{output_dir}/dpo_data_hf.json", "w") as f:
        json.dump(to_hf_dpo_format(dpo_data), f, ensure_ascii=False, indent=2)
    # 同时保存元信息
    meta = {
        "sft_count": len(sft_data),
        "dpo_count": len(dpo_data),
        "sft_sources": {},
        "dpo_sources": {}
    }
    for s in sft_data:
        src = s.get("source", "unknown")
        meta["sft_sources"][src] = meta["sft_sources"].get(src, 0) + 1
    for p in dpo_data:
        src = p.get("source", "unknown")
        meta["dpo_sources"][src] = meta["dpo_sources"].get(src, 0) + 1
    with open(f"{output_dir}/metadata.json", "w") as f:
        json.dump(meta, f, ensure_ascii=False, indent=2)
    print(f"已保存 {len(sft_data)} 条 SFT 数据, {len(dpo_data)} 条 DPO 数据")

Step 6: 质量验证

数据导出之前，我们做最后一关质量验证：

# validate_data.py
def validate_dataset(dataset_name, samples):
    """全面的数据质量验证"""
    issues = []
    total = len(samples)

    # 1. 空值检查
    empty_fields = sum(1 for s in samples
                       if not s.get("instruction", "") and not s.get("prompt", ""))
    if empty_fields:
        issues.append(f"⚠ 空prompt样本: {empty_fields}/{total}")

    # 2. 长度分布
    lengths = [len(s.get("instruction", "") + s.get("output", ""))
               for s in samples]
    avg_len = sum(lengths) / len(lengths)
    issues.append(f"📊 平均样本长度: {avg_len:.0f} 字符")
    issues.append(f"📊 最小长度: {min(lengths)}, 最大长度: {max(lengths)}")

    # 3. 重复率估计
    prompts = [s.get("instruction", "") or s.get("prompt", "") for s in samples]
    unique_prompts = len(set(prompts))
    dup_rate = (total - unique_prompts) / total * 100
    if dup_rate > 5:
        issues.append(f"⚠ 提示词重复率: {dup_rate:.1f}%")
    else:
        issues.append(f"✅ 提示词重复率: {dup_rate:.1f}%")

    # 4. 语言分布
    import re
    chinese_count = sum(1 for p in prompts if re.search(r'[一-鿿]', p))
    cn_ratio = chinese_count / total * 100
    issues.append(f"📊 中文prompt占比: {cn_ratio:.1f}%")

    return issues

完整代码清单

以下是串联所有步骤的主脚本。将上述各步骤的代码整合后，运行这个脚本就能产出一条完整的后训练数据流水线：

# run_pipeline.py — 后训练数据流水线主控脚本
# 使用方式: python run_pipeline.py

import json
import os

# 确保各模块可导入
from download_datasets import download_all
from clean_and_dedup import clean_and_dedup_all
from mix_and_sample import mix_datasets
from build_preference import build_all_dpo
from convert_format import save_datasets
from validate_data import validate_and_report

def main():
    print("=" * 60)
    print("  后训练数据流水线 v1.0")
    print("=" * 60)

    # Step 1: 下载
    print("\n[Step 1/6] 下载数据集...")
    raw_data = download_all()

    # Step 2: 清洗
    print("\n[Step 2/6] 数据清洗与去重...")
    cleaned_data = clean_and_dedup_all(raw_data)

    # Step 3: 配比混合
    print("\n[Step 3/6] 配比混合...")
    sft_mixed = mix_datasets(cleaned_data, target_size=80000)

    # Step 4: 偏好构造
    print("\n[Step 4/6] 偏好数据构造...")
    dpo_pairs = build_all_dpo(raw_data)

    # Step 5: 格式转换
    print("\n[Step 5/6] 格式转换与导出...")
    save_datasets(sft_mixed, dpo_pairs)

    # Step 6: 验证
    print("\n[Step 6/6] 质量验证...")
    validate_and_report(sft_mixed, dpo_pairs)

    print("\n" + "=" * 60)
    print("  流水线完成！最终数据位于 data/final/")
    print("=" * 60)

if __name__ == "__main__":
    main()
实操注意事项：以上代码在生产环境中使用时，建议做以下增强：（1）添加断点续传功能——每个步骤完成后将中间结果存盘，这样某个步骤失败后不需要重新从头开始；（2）添加并行处理——数据清洗和去重可以用多进程加速；（3）添加日志系统——记录每个步骤的耗时和产出统计，方便排查问题；（4）添加数据版本管理——每次运行生成一个版本号，方便追踪不同版本的数据效果差异。
