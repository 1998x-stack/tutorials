第二十八章：端到端RAG项目实战

大家好，我是你们的讲师。欢迎来到本课程的终极篇章！经过前面27章的学习，你已经掌握了RAG的每一个技术环节。今天，我们要做一件有成就感的事——把所有这些知识整合起来，从零开始构建一个真正的企业级知识库问答系统。

让我们设定一个真实的项目背景：某中型科技公司有超过50,000份内部技术文档、产品手册和故障排查指南，分布在Wiki、Confluence和PDF文件中。工程师们每天花费大量时间搜索文档解决问题。我们的目标是用RAG构建一个智能问答助手，让工程师用自然语言提问，在5秒内获得准确的答案，并且每条回答都要标注信息来源。

2. 数据管线搭建

数据管线是整个RAG系统的"水源"。我们从多个数据源摄入文档，经过清洗、分块、嵌入、索引，最终存入向量数据库：

import os
from pathlib import Path
from llama_index.core import (
    SimpleDirectoryReader, VectorStoreIndex, Settings
)
from llama_index.core.node_parser import (
    SentenceSplitter, SemanticSplitterNodeParser
)
from llama_index.embeddings.huggingface import HuggingFaceEmbedding
from llama_index.vector_stores.milvus import MilvusVectorStore
from llama_index.core.ingestion import IngestionPipeline
from llama_index.core.extractors import TitleExtractor

# 1. 配置Embedding模型
Settings.embed_model = HuggingFaceEmbedding(
    model_name="BAAI/bge-large-zh-v1.5",
    trust_remote_code=True
)

# 2. 配置分块策略（针对技术文档优化）
splitter = SentenceSplitter(
    chunk_size=512,
    chunk_overlap=100,
    separator="\n",
    paragraph_separator="\n\n",
)

# 3. 配置向量存储
vector_store = MilvusVectorStore(
    uri="http://milvus:19530",
    collection_name="enterprise_kb",
    dim=1024,        # bge-large-zh维度
    overwrite=False,
)

# 4. 构建数据摄入管线
pipeline = IngestionPipeline(
    transformations=[
        splitter,
        TitleExtractor(),  # 自动提取文档标题作为元数据
        Settings.embed_model,
    ],
    vector_store=vector_store,
)

# 5. 批量摄入文档
def ingest_documents(data_dir: str):
    """从目录中读取并摄入所有支持的文档"""
    reader = SimpleDirectoryReader(
        input_dir=data_dir,
        recursive=True,
        required_exts=[".pdf", ".md", ".docx", ".html", ".txt"],
        file_metadata=lambda x: {
            "source": str(x),
            "ingested_at": str(datetime.now())
        }
    )
    documents = reader.load_data()
    print(f"加载了 {len(documents)} 个文档")

    nodes = pipeline.run(documents=documents)
    print(f"生成 {len(nodes)} 个节点，已存入向量数据库")
    return nodes

# 执行摄入
if __name__ == "__main__":
    ingest_documents("/data/enterprise-docs")

3. 检索与生成集成

数据到位后，核心的检索+生成管线是我们的"大脑"：

from llama_index.core.retrievers import VectorIndexRetriever
from llama_index.core.query_engine import RetrieverQueryEngine
from llama_index.core.postprocessor import SimilarityPostprocessor
from llama_index.llms.openai import OpenAI
from llama_index.core.response_synthesizers import TreeSummarize
import logging

logger = logging.getLogger(__name__)

class EnterpriseRAG:
    """企业知识库RAG问答引擎"""

    def __init__(self, vector_store, llm):
        self.vector_store = vector_store
        self.llm = llm

        # 构建索引
        self.index = VectorStoreIndex.from_vector_store(
            vector_store=vector_store
        )

        # 配置检索器：混合检索（稠密 + 稀疏）
        self.retriever = VectorIndexRetriever(
            index=self.index,
            similarity_top_k=10,  # 初始检索10条
            vector_store_query_mode="hybrid",
        )

        # 配置后处理器：相关性过滤 + 重排序
        self.postprocessor = SimilarityPostprocessor(
            similarity_cutoff=0.75  # 过滤低相关结果
        )

        # 构建查询引擎
        self.query_engine = RetrieverQueryEngine.from_args(
            retriever=self.retriever,
            node_postprocessors=[self.postprocessor],
            llm=self.llm,
            response_mode="tree_summarize",
            verbose=True,
        )

    def query(self, user_query: str, user_role: str = "engineer"):
        """执行一次RAG查询"""
        logger.info(f"用户查询 [{user_role}]: {user_query}")

        # 1. 语义缓存查找（略）

        # 2. 查询重写：将问题转化为检索友好的形式
        rewritten = self._rewrite_query(user_query)

        # 3. 执行检索 + 生成
        response = self.query_engine.query(rewritten)

        # 4. 提取引用来源
        citations = []
        for node in response.source_nodes:
            citations.append({
                "source": node.metadata.get("source", ""),
                "score": round(node.score, 3),
                "text_snippet": node.text[:150] + "..."
            })

        # 5. 事实核查（使用独立LLM调用）
        verification = self._fact_check(user_query, response.response)

        result = {
            "question": user_query,
            "answer": response.response,
            "citations": citations,
            "verification": verification,
            "latency_ms": response.metadata.get("total_latency", 0)
        }

        logger.info(f"查询完成，共 {len(citations)} 个引用")
        return result

    def _rewrite_query(self, query: str) -> str:
        """将用户问题重写为检索友好的形式"""
        rewrite_prompt = f"""将以下技术问题改写为适合文档检索的搜索查询。
保持技术术语不变，去除冗余表达。

原始问题：{query}

搜索查询："""
        response = self.llm.complete(rewrite_prompt)
        return response.text.strip()

    def _fact_check(self, query: str, answer: str) -> dict:
        """对生成结果进行事实核查"""
        check_prompt = f"""请判断以下回答是否完全基于提供的知识库信息。
如果回答中包含知识库之外的信息或推测，请标注。

问题：{query}
回答：{answer}

核查结果："""
        response = self.llm.complete(check_prompt)
        return {
            "passed": "无推测" in response.text or "完全基于" in response.text,
            "detail": response.text
        }

4. 评估与持续优化

系统上线后，真正的挑战才刚开始——持续的评估和优化。我们建立了一个自动化评估管线：

# evaluation_pipeline.py
from ragas import evaluate
from ragas.metrics import (
    faithfulness, answer_relevancy,
    context_precision, context_recall
)
from datasets import Dataset
import json

class RAGEvaluator:
    """RAG系统自动化评估"""

    def __init__(self, rag_system, test_dataset_path: str):
        self.rag = rag_system
        self.test_dataset = self._load_test_dataset(test_dataset_path)

    def _load_test_dataset(self, path):
        """加载标注好的测试集"""
        with open(path) as f:
            data = json.load(f)
        # 格式：[{"question": "...", "ground_truth": "...", "expected_contexts": [...]}]
        return data

    def run_evaluation(self, sample_size: int = 50):
        """运行评估并返回指标"""
        results = []
        for item in self.test_dataset[:sample_size]:
            # 执行RAG查询
            result = self.rag.query(item["question"])

            results.append({
                "question": item["question"],
                "answer": result["answer"],
                "contexts": [c["text_snippet"] for c in result["citations"]],
                "ground_truth": item.get("ground_truth", ""),
                "expected_contexts": item.get("expected_contexts", []),
                "latency": result["latency_ms"]
            })

        # 使用RAGAS指标评估
        dataset = Dataset.from_list(results)
        scores = evaluate(
            dataset,
            metrics=[
                faithfulness,
                answer_relevancy,
                context_precision,
                context_recall
            ]
        )

        return {
            "faithfulness": scores["faithfulness"],
            "relevancy": scores["answer_relevancy"],
            "context_precision": scores["context_precision"],
            "context_recall": scores["context_recall"],
            "avg_latency_ms": sum(r["latency"] for r in results) / len(results),
            "sample_count": len(results)
        }

    def compare_strategies(self):
        """对比不同检索策略的效果"""
        strategies = ["dense", "sparse", "hybrid"]
        results = {}
        for strategy in strategies:
            # 切换检索策略
            self.rag.retriever.vector_store_query_mode = strategy
            results[strategy] = self.run_evaluation(sample_size=30)
        return results

5. 项目经验教训总结

回顾整个项目过程，以下是我们从真实交付中总结的关键教训：

领域 我们踩过的坑 正确的做法 节省的工作量
分块策略 对所有文档使用统一chunk_size=256 按文档类型差异化分块（技术文档chunk=512，FAQ chunk=128） 召回率+15%
Embedding 使用通用英文Embedding处理中文文档 bge-large-zh-v1.5或m3e-large专门处理中文 检索精度+22%
检索策略 只用稠密检索 混合检索（稠密+BM25）配合RFF融合 召回率+18%
评估方式 靠人工抽查 建立1000条标注测试集 + 自动化RAGAS评估 迭代速度提升5倍
缓存 没有缓存 语义缓存 + 精确匹配二级缓存 成本降低40%
监控 没有监控 Prometheus + Grafana全链路监控 MTTR从2小时降到10分钟

6. 持续优化循环

RAG系统不是"一次上线、永久运行"的。我们建立了持续优化循环：

持续优化循环（每两周一次）：

第1天:  收集线上日志 → 抽取低质量回答样本
第2天:  人工标注失败模式（检索失败/生成幻觉/格式错误）
第3天:  针对性改进（调参数/改提示词/加数据）
第4天:  在标注测试集上运行自动化评估
第5天:  对比基线指标，确认改进是否有效
第6天:  如果有效 → 灰度上线；如果无效 → 回退并分析原因

我们从最基础的概念出发——什么是RAG、为什么需要RAG、它和传统搜索的区别是什么。然后我们一步步深入每个技术环节：文档解析、分块策略、向量嵌入、稠密检索、稀疏检索、混合检索、重排序、查询优化、提示工程、引用溯源。我们探讨了幻觉检测、安全护栏、数据隐私这些确保系统可靠性的关键话题。我们学习了评估体系、多模态RAG、GraphRAG、Agentic RAG这些进阶方向。最后，我们比较了主流框架、优化了向量数据库、建立了可观测性体系，并部署到了生产环境。

🚀 最后给你三个建议：

1. 从小处着手：读完这门课后，不要试图一次性构建完美的RAG系统。选一个你最熟悉的领域、拿100份文档、用一个简单但完整的管线跑通全流程。先跑通，再优化。

2. 数据优先：RAG系统的上限由数据质量决定。花最多的时间在数据清洗、分块策略和Embedding质量上——这些环节的投入产出比远高于花时间调提示词。

3. 持续学习：RAG领域的发展速度极快。这门课程教给你的是原理和框架，具体的技术选型和最佳实践半年后可能就会变化。保持好奇，持续跟进。

记住：每一个RAG专家都是从第一个不完美的原型开始的。你的第一个系统可能只有100条文档，检索结果可能不太准确，生成的内容可能还有幻觉——这都没关系。重要的是你开始了，并且知道如何持续改进。
